                if(tab_reviews.length) tab_reviews[0].up().remove();    // If there is a tab remove it

            var relatedAvailable   = $('gallery-film-info-related');

                var tab_related = $$(".tab-related");

            if(!galleryAvailable){
                var tab_gallery = $$(".tab-gallery");
                if(tab_gallery.length) tab_gallery[0].up().remove();
            }

            	silentTriggers: true,
                shouldAnimateOpacityAndHeight: true,
				shouldAnimateContentChange: true,
                initialId: initialSection
            });

			var links = $$('.link-arrow');
			links.each(function(value, key){
				value.target = "_blank";
			});

            contentSwap.setDelegate({
            	isContentLoaded: function(swapView, content){

                    if (content.id === "gallery-film-info-photos" && !content.galleryOverlay ) {
                        return false;
                    }
                    return true;
                },
                loadContent: function(swapView, content){
                    content.galleryOverlay = new OverlayImageGallery(['overlayLink'], {
                    	callBack: function(){
                    		swapView.setLoadedContent(content);
                    	},
                    	incoming: content
                    });
                 },
                didShow: function(swapView, outgoing, incoming){
                	// If the incoming swap is related content, create a scrollbar for it
                	// - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -  //
                    if (incoming.id === "gallery-film-info-related" && !incoming.contentSlider ) {

                    	// Get scroll content available
                    	incoming.scrollContent 	= $(incoming.content).down('.scroll-content');

                    	if(incoming.scrollContent){
	                    	incoming.scrollCount 	= incoming.scrollContent.select('li').length;
	                    	incoming.relatedContent = incoming.scrollContent.down('.related-content');

	                    	var scrollSize 		= (140*incoming.scrollCount),
	                    		containerWidth 	= incoming.scrollContent.getWidth();

	                    	if(scrollSize > containerWidth){

	                    		var handle 		= $(Builder.node('div', {'id':'relatedHandle', 'class':'scroll-handle'}));
	                    		var track 		= $(Builder.node('div', {'id':'relatedTrack',  'class':'scroll-track'}, [handle]));

								incoming.scrollContent.up().appendChild(track);
								handle.style['width'] = containerWidth-(scrollSize-containerWidth) +'px';
		                    	incoming.relatedContent.style['width'] = scrollSize + 'px';

		                    	incoming.scrollHorizontal = function(value, element, slider) {
									element.scrollLeft = Math.round(value/slider.maximum*(element.scrollWidth-element.offsetWidth));
								};

		                        incoming.contentSlider = new Control.ContentSlider('relatedHandle', 'relatedTrack', {
									onSlide: function(v) { incoming.scrollHorizontal(v, incoming.scrollContent, incoming.contentSlider);  },
									onChange: function(v) { incoming.scrollHorizontal(v, incoming.scrollContent, incoming.contentSlider); }
								});
		                    }
		                }
                    }

	                    	incoming: incoming
