   (function($){
        var _grid = "(min-width: 800px)";

        var trigger = '#marvel_cinematic_universe|mouseenter';
        trigger = trigger.split('|');

        var touchShowHideToggle = 0;

        var $module = $('#marvel_cinematic_universe-dropdown'),
            pwrSldr = $module.find('.pwrSldr').data('pwrSldr'),
            $menu_el = $(trigger[0]),
            action_timeout;

        function __init(){
            _bindEvents();
            _calcHidden();
        }

        function _start_hide(){
            if(window.matchMedia(_grid).matches){
                clearTimeout(action_timeout);
                action_timeout = setTimeout(_hide, 400);
            }
        }

        function _show(){
            if(window.matchMedia(_grid).matches){
                clearTimeout(action_timeout);
                touchShowHideToggle = 1;
                $menu_el.find('.dropdown').hide();
                $module.fadeIn(350);
                pwrSldr.update( true );
                $menu_el.addClass('on');
            }
        }

        function _hide(){
            if(window.matchMedia(_grid).matches){
                touchShowHideToggle = 0;
                $module.fadeOut(200);
                $menu_el.removeClass('on').find('.dropdown').css('display','');
            }
        }

        function _bindEvents(){
            //bind events
            $module.on('mouseleave', _start_hide);
            $module.on('mouseenter', _show);

            //bind custom event to menu item
            $menu_el.on(trigger[1], _show);
            $menu_el.on('mouseleave', _start_hide);

            //bind touchEvents - Only for large enough viewports
            if('ontouchstart' in document.documentElement){
                $menu_el.on('touchstart',function(){
                    if(window.matchMedia(_grid).matches)touchShowHideToggle === 0 ? _show() : _hide();
                    else {$menu_el.removeClass('off');}
                });
            }
        }

        function _calcHidden(mod){
            var m = $module;
            if(m.not(':visible')){
                m.css({'display' : 'block', 'visibility' : 'hidden'});
                pwrSldr.update( true );
                m.one('updatedDimensions',function(){
                    m.css({'display' : '', 'visibility' : ''});
                });
            }
         }

        __init();

    })(jQuery);
