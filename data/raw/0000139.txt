            });

				value.target = "_blank";
			});

                        return false;
                    }
                    return true;
                },
                loadContent: function(swapView, content){
                    content.galleryOverlay = new OverlayImageGallery(['overlayLink'], {
                    	callBack: function(){
                    		swapView.setLoadedContent(content);
                    	},
                    	incoming: content
                    });
                 },
                didShow: function(swapView, outgoing, incoming){
                	// If the incoming swap is related content, create a scrollbar for it
                	// - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -  //
                    if (incoming.id === "gallery-film-info-related" && !incoming.contentSlider ) {

                    	// Get scroll content available
                    	incoming.scrollContent 	= $(incoming.content).down('.scroll-content');

                    	if(incoming.scrollContent){
	                    	incoming.scrollCount 	= incoming.scrollContent.select('li').length;
	                    	incoming.relatedContent = incoming.scrollContent.down('.related-content');

	                    	var scrollSize 		= (140*incoming.scrollCount),

									element.scrollLeft = Math.round(value/slider.maximum*(element.scrollWidth-element.offsetWidth));
