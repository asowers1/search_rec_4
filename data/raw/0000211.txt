                        return { name: data.name, value: data, query: searchLoading.query() };
                    });
                },
                ajax: {
                    beforeSend: function() {
                        searchLoading.loadingData();
                    },
                    complete: function() {
                        searchLoading.stopLoadingData();
                    }
                }
            },
            limit:2
        }),
        tvResults = new Bloodhound({
            datumTokenizer: Bloodhound.tokenizers.obj.whitespace('value'),
            queryTokenizer: Bloodhound.tokenizers.whitespace,
            remote: {
                url: "/search/json/?catCount=2&q=",
                replace: function(url, query) {
                    searchLoading.updateQuery(query);
                    return url + query;
                },
                filter: function(media) {
                    return $.map(media.tvResults, function(data) {
                        data.type = "tv";
                        return { name: data.name, value: data, query: searchLoading.query() };
                    });
                }
            },
            limit:2
        }),
        actorResults = new Bloodhound({
            datumTokenizer: Bloodhound.tokenizers.obj.whitespace('value'),
            queryTokenizer: Bloodhound.tokenizers.whitespace,
            remote: {
                url: "/search/json/?catCount=2&q=",
                replace: function(url, query) {
                    searchLoading.updateQuery(query);
                    return url + query;
                },
                filter: function(media) {
                    return $.map(media.actors, function(data) {
                        data.type = "actor";
                        return { name: data.name, value: data, query: searchLoading.query() };
                    });
                }
            },
            limit:2
        });

    movieResults.initialize();
    tvResults.initialize();
    actorResults.initialize();

    $("#search-term").typeahead({
        highlight:true,
        minLength:2
    }, {
        name: "movies",
        source: movieResults.ttAdapter(),
        displayKey: function(data) {
            return data.name;
        },
        templates: {
            header: "<h3>Movies</h3>",
            suggestion: _.template("<div class='searchResult movieResult'><a class='media' href='/m/<@= value.vanity @>/?search=<@= query @>'><div class='pull-left searchResultImage'><img class='media-object' src='<@= value.image @>'/></div><div class='media-body'><p class='media-heading linkColor noSpacing small'><@= value.name @> (<@= value.year @>)</p><p class='small subtle noSpacing'><@= value.subline @></p></div></a></div>"),
            footer: _.template("<p class='more pull-right viewAllSearchResults'><a href='/search/?search=<@= query @>'>View All Search Results <span class='fontelloIcon icon-angle-right'></span></a></p>")
        }
    }, {
        name: "tv",
        source: tvResults.ttAdapter(),
        displayKey: function(data) {
            return data.name;
        },
        templates: {
            header: "<h3>TV Series</h3>",
            suggestion: _.template("<div class='searchResult tvResult'><a class='media' href='/tv/<@= value.vanity @>/?search=<@= query @>'><div class='pull-left searchResultImage'><img class='media-object' src='<@= value.image @>'/></div><div class='media-body'><p class='media-heading linkColor noSpacing small'><@= value.name @> (<@= value.startYear @> - <@= value.endYear @>)</p><p class='small subtle noSpacing'><@= value.subline @></p></div></a></div>"),
            footer: _.template("<p class='more pull-right viewAllSearchResults'><a href='/search/?search=<@= query @>'>View All Search Results <span class='fontelloIcon icon-angle-right'></span></a></p>")
        }
    }, {
        name: "actors",
        source: actorResults.ttAdapter(),
        displayKey: function(data) {
            return data.name;
        },
        templates: {
            header: "<h3>Celebrities</h3>",
            suggestion: _.template("<div class='searchResult celebrityResult'><a class='media' href='/celebrity/<@= value.vanity @>/?search=<@= query @>'><div class='pull-left searchResultImage'><img class='media-object' src='<@= value.image @>'/></div><div class='media-body'><p class='media-heading linkColor noSpacing small'><@= value.name @></p></div></a></div>"),
            footer: _.template("<p class='more pull-right viewAllSearchResults'><a href='/search/?search=<@= query @>'>View All Search Results <span class='fontelloIcon icon-angle-right'></span></a></p>")
        }
    }).on("typeahead:selected", function(evt, data) {
        debugger;
        if ("movie" == data.value.type) {
            document.location.href = "/m/" + data.value.vanity + "/?search=" + searchLoading.query();
        } else if ("tv" == data.value.type) {
            document.location.href = "/tv/" + data.value.vanity + "/?search=" + searchLoading.query();
        } else {
            document.location.href = "/celebrity/" + data.value.vanity + "/?search=" + searchLoading.query();
        }
    }).on("typeahead:opened", function() {
        $(".twitter-typeahead").css("position", "static");
        $(".twitter-typeahead .tt-input").css({"position":"absolute","width":$(".twitter-typeahead").width()});
        $(".twitter-typeahead .tt-hint").css("position", "relative");
        $(".tt-dropdown-menu").css("top","36px");
    });
    var onResize = function() {
        $(".twitter-typeahead").css("width", $(".navbar-searchbox").width()-$(".navbar-searchbox .input-group-btn").width());
        $(".twitter-typeahead .tt-input").css({"width":$(".twitter-typeahead").width()});
    };
    $(window).resize(onResize);
    $(window).on("orientationchange", onResize);
    $(".header_main .navbar-searchbox .searchBtn").click(function() {
        var searchQuery = $(".header_main .navbar-searchbox .twitter-typeahead pre").text();
        searchQuery = searchQuery.replace(/\ /g, "+");
        window.location.href = "/search/?search="+searchQuery;
    });
    
    $("[rel=tooltip], [data-toggle=tooltip]").tooltip();
});
